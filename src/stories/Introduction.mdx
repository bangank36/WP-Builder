## ðŸš€ Getting Started

Install the library and peer-dependencies in your application

Pass the JSON Schema to the `JsonForms` component.

```jsx
import { JsonForms } from "@jsonforms/react";
import { vanillaCells, vanillaRenderers } from '@jsonforms/vanilla-renderers';
import { enumExample } from '@jsonforms/examples'; 
import React from 'react';

export const MyForm = () => {
  const [data, setData] = useState(initialData);

  return <JsonForms
          schema={exampleData.schema}
          uischema={exampleData.uischema}
          data={exampleData.data}
          renderers={vanillaRenderers}
          cells={vanillaCells}
          onChange={({ data, _errors }) => {
            setData(data);
          }}
        />
}
```

For further examples see the stories.
